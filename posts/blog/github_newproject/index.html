    Github批量创建项目#

    1. 功能背景#

    AOSP的代码是使用repo管理的，需要上传至github上的时候需要批量创建项目。

    2. 功能构思#

    从源代码库的.repo/manifests/default.xml文件中读取出来每一个项目的name及path，然后使用github的api批量创建项目，这里要注意的是qcom提供的代码是使用斜杠分开的，而我们在github中是使用了类似lineageos中的做法将斜杠用下划线代替。

    3. fuck code#

    createTask(){
    #system/tools/mkbootimg : LineageOS/android_system_tools_mkbootimg
    line=$1
    prj_name=${line##*:}
    prj_name=${prj_name//\//_}
    prj_name=`eval echo $prj_name`
    while true;
    do
        index=$[RANDOM%${#PRI_TOKEN[@]}]
        token=${PRI_TOKEN[$index]}
        resp=`curl -s -X POST -H "Accept: application/vnd.github+json" -H "Authorization: Bearer $token" -d "{\"name\":\"$prj_name\",\"private\":\"true\",\"has_issues\":\"fasle\",\"has_projects\":\"true\",\"has_wiki\":\"false\",\"auto_init\":\"false\"}" https://api.github.com/orgs/$org/repos`
        if [[ $resp == *"secondary rate limit"* ]]; then
            echo -e "$token : $prj_name -----> rate limit"
            sleep 15s
        else
            msg=`echo $resp | jq '.url'`
            if [ "$msg" == "null" ];then
                msg=`echo $resp | jq '.errors[0].message'`
            fi
            echo -e "$token : $prj_name ----->  $msg"
            sleep 5s
            break
        fi
    done
}

syncCreate(){
    IFS=$'\n'
    for line in $repolist;do
        createTask $line
        #break;
    done    
}

    4. 多线程push代码#

    pushTask(){
    #system/tools/mkbootimg : LineageOS/android_system_tools_mkbootimg
    line=$1
    prj_path=${line%%:*}
    prj_path=`eval echo $prj_path`
    prj_name=${line##*:}
    #prj_name=${prj_name#*LineageOS/}
    prj_name=${prj_name//\//_}
    prj_name=`eval echo $prj_name`
    cd $prj_path
    REMOTE=git@github.com:$org/${prj_name}
    cmd="git push --porcelain --all $REMOTE"
    if [ -n "$branch" ]; then
        cmd="git push --porcelain $REMOTE HEAD:refs/heads/$branch"
    fi
    pushres=`eval $cmd`
    if [[ $pushres == *"Done"* ]]; then
        echo "$cmd  ---> succed"
    else
        echo "$cmd ---> failed $pushres"
        echo "$prj_path:$prj_name" >>$logpath
    fi
    cd - >/dev/null
}

asyncPush(){
    fifofile="/tmp/$$.fifo"
    mkfifo $fifofile
    exec 6<>$fifofile
    rm -rf $fifofile
    thread_num=10
    job_num=${#repolist[@]}
    for ((i=0;i<${thread_num};i++)); do
        echo >&6
    done
    rm -rf $logpath
    IFS=$'\n'
    for line in $repolist;do
    {
        read -u6
        pushTask $line
        echo >&6
    }&
    done
    wait
    exec 6<&-
    exec 6>&-
    echo "sync end"
}

    5. 输出新的manifest.xml文件#

    toXml(){
    rm -rf manifest_new.xml
    while read line
    do
        pline=$line
        if [[ $line == \<project* ]];then
            #echo $line
            prj_name=${line#*name=\"}
            prj_name=${prj_name%%\"*}
            prj_path=${line#*path=\"}
            prj_path=${prj_path%%\"*}
            end=${line##*\"}
            pline="<project name=\"$prj_name\" path=\"$prj_path\" $end"
        fi
        echo $pline >> manifest_new.xml
    done < .repo/manifests/default.xml
}

    6. 后记#

    github创建项目如果过快会触发secondary rate limit 因此，可以多拿几个token随机使用创建项目，而gitlab却没有这种限制。
