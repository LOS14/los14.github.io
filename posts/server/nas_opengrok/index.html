    NAS上搭建Opengrok代码搜索引擎#

    一. 功能背景#

    1. 使用opengrok搭建多项目搜索
    2. opengrok-1.1-rc34
    3. ubuntu18.04 + tomcat8 + ctags + openjdk-8

    二. 自动创建、同步脚本#

    #!/bin/bash
LOCAL_PATH=`pwd`
WEB_APP="/var/lib/tomcat8/webapps"

createPrj(){
    export OPENGROK_GENERATE_HISTORY=0
    export OPENGROK_INSTANCE_BASE=$LOCAL_PATH/index/$1
    export OPENGROK_WEBAPP_NAME=$1
    export OPENGROK_WEBAPP_CONTEXT=$1
    export OPENGROK_WEBAPP_CFGADDR=http://localhost:8080/$1
    ./bin/OpenGrok deploy
}

indexPrj(){
    ./bin/OpenGrok index $LOCAL_PATH/database/$1
}

rootHtml(){
    projects=""
    indexInfo=""
    html=$(cat "./ROOT/index.html")
    for dir in `ls $WEB_APP`;do
        if [[ -d "$WEB_APP/$dir" ]] && [[ "ROOT" != $dir ]]; then
            projects="<li><a href=\"/$dir/\">$dir</a></li>\n""$projects"
            indexLog="./index/$dir/log/opengrok0.0.log"
            lastIndex=`stat --format=%y $indexLog`
            lastIndex=${lastIndex%.*}
            indexInfo="<li><b>$lastIndex</b> - New Index: <a href=\"/$dir/\">$dir</a></li>\n""$indexInfo"
       fi
    done
    html=${html//projects/$projects}
    html=${html//project_index/$indexInfo}
    rm -rf $WEB_APP/ROOT
    mkdir -p $WEB_APP/ROOT
    echo -e "$html" >$WEB_APP/ROOT/index.html
    cp ./ROOT/style.css $WEB_APP/ROOT/style.css
    rm -rf $WEB_APP/*.war
}

main(){
    for dir in `ls database`;do
        rm -rf $database/$dir/{prebuilts,developers,development}
        if [[ $1 == "create" ]];then
            createPrj $dir
        fi
        indexPrj $dir
    done
    rootHtml
    /etc/init.d/tomcat8 restart
}
main $*

    三. 后记#

    本功能在于opengrok的版本选择，opengrok-1.1-rc34版本正好简单安装在ubuntu18.04上，不追求最新版，只求最稳定
